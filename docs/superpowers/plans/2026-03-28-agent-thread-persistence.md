# Agent Thread Persistence Implementation Plan

> **For agentic workers:** REQUIRED SUB-SKILL: Use superpowers:subagent-driven-development (recommended) or superpowers:executing-plans to implement this plan task-by-task. Steps use checkbox (`- [ ]`) syntax for tracking.

**Goal:** Persist agent chat sessions in SQLite so users can create multiple named threads per database connection and resume them after app restart.

**Architecture:** SDK JSONL files at `~/.config/tables/sessions/{threadId}/` handle API conversation context (resume, compaction). SQLite `agent_threads`, `agent_messages`, and `agent_tool_calls` tables store display cache and metadata. The harness sidecar captures `sdk_session_id` from the SDK's init event and emits it to the frontend via SSE; the frontend saves it to `agent_threads` for future resumes.

**Tech Stack:** Rust/rusqlite (sync, Mutex-wrapped), TypeScript/Svelte 5 runes, Bun/SSE harness, `@anthropic-ai/claude-agent-sdk` v0.2.84

```sql
-- Migration: 007_create_agent_tables
-- Purpose: Persist agent chat threads, display messages, and tool calls

CREATE TABLE IF NOT EXISTS agent_threads (
    id TEXT PRIMARY KEY,                    -- UUID, generated by frontend
    title TEXT NOT NULL DEFAULT 'New chat', -- Auto-set from first user message
    connection_id TEXT NOT NULL,            -- Connection this thread belongs to
    database_name TEXT,                     -- Selected database at thread creation
    model TEXT NOT NULL,                    -- e.g. "claude-sonnet-4-6"
    effort TEXT NOT NULL DEFAULT 'auto',    -- auto | low | medium | high | max
    sdk_session_id TEXT,                    -- SDK JSONL session UUID (set after first turn)
    summary TEXT,                           -- Post-compaction summary (future use)
    created_at INTEGER NOT NULL,
    updated_at INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS agent_messages (
    id TEXT PRIMARY KEY,                    -- UUID
    thread_id TEXT NOT NULL REFERENCES agent_threads(id) ON DELETE CASCADE,
    role TEXT NOT NULL,                     -- "user" | "assistant"
    content TEXT NOT NULL DEFAULT '',
    thinking TEXT,                          -- Extended thinking content, if any
    timestamp INTEGER NOT NULL
);

CREATE TABLE IF NOT EXISTS agent_tool_calls (
    id TEXT PRIMARY KEY,                    -- toolId from SDK event
    thread_id TEXT NOT NULL REFERENCES agent_threads(id) ON DELETE CASCADE,
    tool_name TEXT NOT NULL,
    input TEXT NOT NULL DEFAULT '{}',       -- JSON-serialised input
    output TEXT,                            -- NULL while running
    status TEXT NOT NULL DEFAULT 'running', -- running | done | error
    started_at INTEGER NOT NULL,
    completed_at INTEGER
);

CREATE INDEX IF NOT EXISTS idx_agent_threads_connection ON agent_threads(connection_id);
CREATE INDEX IF NOT EXISTS idx_agent_messages_thread ON agent_messages(thread_id);
CREATE INDEX IF NOT EXISTS idx_agent_tool_calls_thread ON agent_tool_calls(thread_id);
```

```typescript
// src/lib/stores/threads.svelte.ts
import { invoke } from "@tauri-apps/api/core";

export interface AgentThread {
    id: string;
    title: string;
    connectionId: string;
    databaseName: string | null;
    model: string;
    effort: string;
    sdkSessionId: string | null;
    summary: string | null;
    createdAt: number;
    updatedAt: number;
}

function nowSecs(): number {
    return Math.floor(Date.now() / 1000);
}

class ThreadsStore {
    threads = $state<AgentThread[]>([]);
    activeThreadId = $state<string | null>(null);

    get activeThread(): AgentThread | null {
        return this.threads.find((t) => t.id === this.activeThreadId) ?? null;
    }

    async load(connectionId: string, databaseName: string | null) {
        try {
            const rows = await invoke<AgentThread[]>("list_agent_threads", {
                connectionId,
                databaseName,
            });
            this.threads = rows;
        } catch (e) {
            console.error("[threads] load failed:", e);
            this.threads = [];
        }
    }

    async createThread(opts: {
        connectionId: string;
        databaseName: string | null;
        model: string;
        effort: string;
    }): Promise<AgentThread> {
        const id = crypto.randomUUID();
        const now = nowSecs();
        await invoke("create_agent_thread", {
            id,
            connectionId: opts.connectionId,
            databaseName: opts.databaseName,
            model: opts.model,
            effort: opts.effort,
            now,
        });
        const thread: AgentThread = {
            id,
            title: "New chat",
            connectionId: opts.connectionId,
            databaseName: opts.databaseName,
            model: opts.model,
            effort: opts.effort,
            sdkSessionId: null,
            summary: null,
```

```svelte
        }

        navigator.clipboard.writeText(lines.join("\n")).then(() => {
            copyDone = true;
            setTimeout(() => { copyDone = false; }, 2000);
        });
    }
</script>

                        <IconCheck size={12} class="text-green-500" />
                    {:else}
                        <IconCopy size={12} />
                    {/if}
                </button>
            {/if}
        </div>
    </div>

    <!-- Content -->
    {#if !schemaStore.activeConnection}
        <div class="flex flex-1 flex-col items-center justify-center gap-2 px-4 text-center text-muted-foreground">
            <IconAi size={24} class="opacity-30" />

            <IconAlertCircle size={20} class="text-destructive" />
            <span class="text-[12px] text-muted-foreground">{sessionError}</span>

        </div>
    {:else}
        <MessageList onRunQuery={handleRunQuery} />
        <AgentComposer
            onSend={(displayText, fullText, _doc) => send(displayText, fullText)}
            onStop={stop}
            running={agentStore.status === "running"}
            disabled={!sessionReady || !!sessionError}
        />
    {/if}
```
