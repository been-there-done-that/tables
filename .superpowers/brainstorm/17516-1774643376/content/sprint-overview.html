Track 3 — DDL Tabs + Backend Introspection

CREATE OR REPLACE FUNCTION
public.get_user_by_id(user_id uuid)
RETURNS
TABLE(id uuid, email text, created_at timestamptz)
LANGUAGE
plpgsql
SECURITY DEFINER AS $$
BEGIN
RETURN QUERY
SELECT id, email, created_at
FROM users WHERE id = user_id;
END; $$
